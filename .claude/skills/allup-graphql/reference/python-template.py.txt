def get_token(env: str) -> str:
    """Get admin auth token via wrk CLI."""
    result = subprocess.run(
        ["wrk", "-e", env, "login", "admin", "--print"],
        capture_output=True,
        text=True,
        check=True,
    )
    return result.stdout.strip()


# =============================================================================
# GraphQL Client
# =============================================================================


def execute_query(query: str, variables: dict | None = None) -> dict:
    """Execute a GraphQL query and return the result."""
    token = get_token(ENV)
    response = requests.post(
        API_URLS[ENV],
        headers={
            "Authorization": f"Bearer {token}",
            "Content-Type": "application/json",
        },
        json={"query": query, "variables": variables or {}},
    )
    response.raise_for_status()
    result = response.json()
    if "errors" in result:
        raise Exception(f"GraphQL errors: {json.dumps(result['errors'], indent=2)}")
    return result["data"]


def paginate(query: str, connection_path: str, variables: dict | None = None):
    """
    Iterate through all pages of a paginated query.

    Args:
        query: GraphQL query with $after variable for cursor
        connection_path: Dot-separated path to the connection (e.g., "listApplicants")
        variables: Base variables (will add 'after' automatically)

    Yields:
        Individual nodes from each page

    Example:
        query = '''
            query($after: String) {
                listApplicants(limit: 100, after: $after) {
                    nodes { id status }
                    pageInfo { hasNextPage endCursor }
                }
            }
        '''
        for applicant in paginate(query, "listApplicants"):
            print(applicant["id"])
    """
    variables = dict(variables or {})
    cursor = None

    while True:
        variables["after"] = cursor
        data = execute_query(query, variables)

        # Navigate to the connection
        connection = data
        for key in connection_path.split("."):
            connection = connection[key]

        # Yield nodes
        for node in connection["nodes"]:
            yield node

        # Check for next page
        page_info = connection["pageInfo"]
        if not page_info["hasNextPage"]:
            break
        cursor = page_info["endCursor"]
